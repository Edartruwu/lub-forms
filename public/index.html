Lub Forms

Embeddable form library for Lub CRM

Quick Start - Standalone Script

<!-- Add this to your HTML -->
<script src="https://forms.lub.marketing/lub-forms.standalone.js"></script>

<!-- Add a container with your form ID -->
<div
  data-lub-form-id="YOUR_FORM_ID"
  data-lub-base-url="https://api.lub.marketing"
></div>

Manual Render with Callbacks

<div id="my-form"></div>

<script>
  LubForms.render('YOUR_FORM_ID', 'my-form', {
    baseUrl: 'https://api.lub.marketing',
    onSuccess: (data) => {
      console.log('Submitted!', data.submission_id);
      window.location.href = '/thank-you';
    },
    onError: (err) => {
      console.error('Error:', err);
    }
  });
</script>
